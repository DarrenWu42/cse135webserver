CREATE TABLE `static`( 
    `sess_id` VARCHAR(32) UNIQUE,
    `user_agent` VARCHAR(255),
    `language` VARCHAR(16),
    `cookies` TINYINT(1) UNSIGNED,
    `inner_width` SMALLINT(4) UNSIGNED,
    `inner_height` SMALLINT(4) UNSIGNED,
    `outer_width` SMALLINT(4) UNSIGNED,
    `outer_height` SMALLINT(4) UNSIGNED,
    `downlink` DECIMAL(7,3) UNSIGNED,
    `effective_type` VARCHAR(8),
    `rtt` SMALLINT(4) UNSIGNED,
    `save_data` TINYINT(1) UNSIGNED,
    PRIMARY KEY (`sess_id`)
);

CREATE TABLE `performance`(
	`sess_id` VARCHAR(32) UNIQUE,
	`start_time` TIMESTAMP(3) NULL DEFAULT NULL,
	`fetch_start` TIMESTAMP(3) NULL DEFAULT NULL,
	`request_start` TIMESTAMP(3) NULL DEFAULT NULL,
	`response_start` TIMESTAMP(3) NULL DEFAULT NULL,
	`response_end` TIMESTAMP(3) NULL DEFAULT NULL,
	`dom_interactive` TIMESTAMP(3) NULL DEFAULT NULL,
	`dom_loaded_start` TIMESTAMP(3) NULL DEFAULT NULL,
	`dom_loaded_end` TIMESTAMP(3) NULL DEFAULT NULL,
	`dom_complete` TIMESTAMP(3) NULL DEFAULT NULL,
	`load_event_start` TIMESTAMP(3) NULL DEFAULT NULL,
	`load_event_end` TIMESTAMP(3) NULL DEFAULT NULL,
	`duration` DECIMAL(14, 3) UNSIGNED,
	`transfer_size` MEDIUMINT(6) UNSIGNED,
	`decoded_body_size` MEDIUMINT(6) UNSIGNED,
	PRIMARY KEY (`sess_id`),
	FOREIGN KEY (`sess_id`) REFERENCES static(`sess_id`)
);

CREATE TABLE `activity`(
	`sess_id` VARCHAR(32),
	`start_time` TIMESTAMP(3) DEFAULT NULL,
	`fetch_start` TIMESTAMP(3),
	`request_start` TIMESTAMP(3),
	`response_start` TIMESTAMP(3),
	`response_end` TIMESTAMP(3),
	`dom_interactive` TIMESTAMP(3),
	`dom_loaded_start` TIMESTAMP(3),
	`dom_loaded_end` TIMESTAMP(3),
	`dom_complete` TIMESTAMP(3),
	`load_event_start` TIMESTAMP(3),
	`load_event_end` TIMESTAMP(3),
	`duration` DECIMAL(14, 3) UNSIGNED,
	`transfer_size` MEDIUMINT(6) UNSIGNED,
	`decoded_body_size` MEDIUMINT(6) UNSIGNED,
	PRIMARY KEY (`sess_id`),
	FOREIGN KEY (`sess_id`) REFERENCES static(`sess_id`)
);
